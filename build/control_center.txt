cc_log_in(From, Payload) :-
    format('Control center | IN  from=~w payload=~w~n', [From, Payload]).

cc_send(To, Payload) :-
    clause(agent(CC),_),
    format('Control center | OUT to=~w payload=~w~n', [To, Payload]),
        ( a(message(To, Payload))
        ; ( Payload = inform(Content, _Meta, _From)
            , a(message(To, inform(Content, CC)))
            )
        ; ( Payload = inform(Content, _Meta)
            , a(message(To, inform(Content, CC)))
            )
        ; a(message(To, inform(Payload, CC)))
        ; a(message(To, Payload, CC))
        ), !.

cc_send_collect(Truck, BinID) :-
    clause(agent(CC),_),
    cc_send(Truck, inform(collect(BinID), CC)).

cc_send_cleared(BinID) :-
    clause(agent(CC),_),
    cc_send(BinID, inform(cleared, CC)).

cc_log_status(BinID, Status, Template, Args) :-
    ( clause(cc_last_status(BinID, Status),_) ->
        true
    ; retractall(cc_last_status(BinID,_)),
      assert(cc_last_status(BinID, Status)),
      format(Template, Args)
    ).

schedule_dispatch_delay(BinID) :-

record_timeout(BinID, Count) :-
    ( clause(timeout_count(BinID, C0),_) -> C1 is C0 + 1 ; C1 = 1 ),
    retractall(timeout_count(BinID,_)),
    assert(timeout_count(BinID, C1)),
    Count = C1.

clear_timeout_count(BinID) :-
    retractall(timeout_count(BinID,_)).

tick_inflight_watchdog(dummy) :-
    clause(inflight(BinID),_),
    clause(inflight_countdown(BinID, 0),_),
    record_timeout(BinID, TimeoutN),
    max_dispatch_timeouts(MaxTimeouts),
    ( TimeoutN >= MaxTimeouts ->
        cc_log_status(BinID, fallback(TimeoutN), 'Control center | fallback: forcing completion for bin=~w after ~w timeouts~n', [BinID, TimeoutN]),
        cc_send(logger, inform(completed(timeout, BinID), CC)),
        cc_send_cleared(BinID),
        retractall(awaiting(BinID)),
        retractall(assigned_bin(BinID,_)),
        retractall(bin_ready_for_collection(BinID)),
        clear_timeout_count(BinID)
    ; true ),
    cc_log_status(BinID, timeout(TimeoutN), 'Control center | timeout: no truck response for bin=~w, releasing inflight~n', [BinID]),
    retractall(inflight(BinID)),
    clear_inflight_timer(BinID),
    schedule_retry(BinID, timeout),
    fail.

remove_seen([], _, []).
remove_seen([H|T], Seen, R) :-
    member(H, Seen),
    remove_seen(T, Seen, R).
remove_seen([H|T], Seen, [H|R]) :-
    \+ member(H, Seen),
    remove_seen(T, Seen, R).

truck_busy(Truck) :-
    clause(assigned_bin(_, Truck),_).
truck_busy(Truck) :-
    clause(last_try(BinID, Truck),_),
    clause(inflight(BinID),_).

remove_busy([], []).
remove_busy([H|T], R) :-
    truck_busy(H),
    remove_busy(T, R).
remove_busy([H|T], [H|R]) :-
    \+ truck_busy(H),
    remove_busy(T, R).

random_pick(List, Pick) :-
    length(List, L),
    L > 0,
    random(0, L, N),
    nth0(N, List, Pick).

dispatch_request(BinID) :-
    \+ clause(bin_ready_for_collection(BinID),_),
    cc_log_status(BinID, blocked_not_full, 'Control center | blocked: bin=~w not full yet, collection skipped~n', [BinID]),
    clause(agent(CC),_),
    cc_send(logger, inform(blocked_not_full(BinID), CC)),
    !.

dispatch_request(BinID) :-
    clause(inflight(BinID),_), !.

dispatch_request(BinID) :-
    clause(assigned_bin(BinID,_),_), !.

dispatch_request(BinID) :-
    truck_pool(AllTrucks),
    ( clause(tried_trucks(BinID, Tried),_) -> true ; Tried = [] ),
    remove_seen(AllTrucks, Tried, Candidates0),
    ( Candidates0 = [] ->
        retractall(tried_trucks(BinID,_)),
        Candidates = AllTrucks,
        Retries = cycled
    ; Candidates = Candidates0,
      Retries = normal
    ),
    remove_busy(Candidates, IdleCandidates),
        ( IdleCandidates = [] ->
                cc_log_status(BinID, deferred_busy, 'Control center | deferred: all trucks busy for bin=~w, dispatching with fallback pool~n', [BinID]),
                clause(agent(CC),_),
                cc_send(logger, inform(deferred_busy(BinID), CC))
        ; true ),
    ( IdleCandidates = [] -> PickPool = Candidates ; PickPool = IdleCandidates ),
    random_pick(PickPool, Truck),
    retractall(last_try(BinID,_)),
    assert(last_try(BinID, Truck)),
    retractall(inflight(BinID)),
    assert(inflight(BinID)),
    start_inflight_timer(BinID),
    cc_log_status(BinID, dispatch(Truck, Retries), 'Control center | dispatch: truck=~w, bin=~w, mode=~w~n', [Truck, BinID, Retries]),
    clause(agent(CC),_),
    cc_send(logger, inform(requesting(Truck, BinID, Retries), CC)),
    cc_send_collect(Truck, BinID).
