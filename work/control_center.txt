remove_seen([], _, []).
remove_seen([H|T], Seen, R) :-
    member(H, Seen),
    remove_seen(T, Seen, R).
remove_seen([H|T], Seen, [H|R]) :-
    \+ member(H, Seen),
    remove_seen(T, Seen, R).

random_pick(List, Pick) :-
    length(List, L),
    L > 0,
    random(0, L, N),
    nth0(N, List, Pick).

dispatch_request(BinID) :-
    truck_pool(AllTrucks),
    ( clause(tried_trucks(BinID, Tried),_) -> true ; Tried = [] ),
    remove_seen(AllTrucks, Tried, Candidates0),
    ( Candidates0 = [] ->
        retractall(tried_trucks(BinID,_)),
        Candidates = AllTrucks,
        Retries = cycled
    ; Candidates = Candidates0,
      Retries = normal
    ),
    random_pick(Candidates, Truck),
    retractall(last_try(BinID,_)),
    assert(last_try(BinID, Truck)),
    format('[control_center] Requesting ~w to collect ~w (~w)~n', [Truck, BinID, Retries]),
    clause(agent(CC),_),
    a(message(logger, inform(requesting(Truck, BinID, Retries), CC))),
    a(message(Truck, inform(collect(BinID), CC))).

    normalize_full(Msg, BinID),
    ( clause(awaiting(BinID),_) ->
        true
    ; format('[control_center] FULL alert received for ~w~n', [BinID]),
      assert(awaiting(BinID)),
      retractall(tried_trucks(BinID,_)),
      assert(tried_trucks(BinID, [])),
      clause(agent(CC),_),
      a(message(logger, inform(full_received(BinID), CC))),
      dispatch_request(BinID)
    ).
